# modul-4-operator-
berisi sample koding operator aritmatika dan relasional

# Contoh program
def main():
    # mendefinisikan variabel x dan y
    x = 8
    y = 3

    # menampilkan nilai x dan y
    print('x =', x)
    print('y =', y)

    # melakukan perhitungan terhadap x dan y
    print('x + y', (x + y))
    print('x - y', (x - y))
    print('x * y', (x * y))
    print('x / y', (x / y))
    print('x // y', (x // y))
    print('x % y', (x % y))
    print('x ** y', (x ** y))

if __name__ == '__main__':
    main()
